``` c
#include <stdio.h>

int main(int argc, char const *argv[]) {
	int num[5], *p;
	for (int i = 0; i < 5; i++) {
		printf("Please type in an integer:\n");
		scanf("%d", &num[i]);
	}
	printf("First:\n");
	for (int i = 4; i >= 0; i--) {
		printf("%d\n", num[i]);
	}
	printf("Second:\n");
	for (int i = 0; i >= 0; i--) {
		printf("%d\n", *(num + i));
	}
	p = num;
	printf("Third:\n");
	for (int i = 0; i >= 0; i--) {
		printf("%d\n", p[i]);
	}
	printf("Fourth:\n");
	for (int i = 4; i >= 0; i--) {
		printf("%d\n", *(p + i));
	}
	return 0;
}
```

``` c
#include <stdio.h>

int main(int argc, char const *argv[]) {
	float num[8], sum;
	for (int i = 0; i < 8; i++) {
		printf("Please type in a float number:\n");
		scanf("%f", &num[i]);
	}
	for (int i = 1; i < 8; i += 2) {
		printf("%d: %f\n", i + 1, *(num + i));
	}
	for (int i = 0; i < 8; i += 2) {
		sum += *(num + i);
	}
	printf("Sum is %f", sum);
	return 0;
}
```

``` c
#include <stdio.h>

int main(int argc, char const *argv[]) {
	int num[6], maxmin(int[], int);
	for (int i = 0; i < 6; i++) {
		printf("Please type in an integer:\n");
		scanf("%d", &num[i]);
	}
	maxmin(num, 6);
	return 0;
}

int maxmin(int number[], int n) {
	int max, min;
	max = min = number[0];
	for (int i = 1; i < n; i++) {
		max = max > number[i] ? max : number[i];
		min = min < number[i] ? min : number[i];
	}
	printf("max = %d\nmin = %d", max, min);
	return 0;
}
```
